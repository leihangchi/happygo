                <li>使用"给我提示"按钮可以获得一个有用的提示</li>

    <script>

        // 游戏变量

        let targetNumber;

        let attemptsLeft;

        let attemptsUsed = 0;

        let gameOver = false;

        let difficulty = 'easy';

        let minRange = 1;

        let maxRange = 100;

        let guessHistory = [];

 

        // 根据难度设置尝试次数

        const difficultySettings = {

            easy: 10,

            medium: 7,

            hard: 5

        };

 

        // DOM元素

        const guessInput = document.getElementById('guessInput');

        const submitBtn = document.getElementById('submitBtn');

        const restartBtn = document.getElementById('restartBtn');

        const hintBtn = document.getElementById('hintBtn');

        const attemptsDisplay = document.getElementById('attempts');

        const usedDisplay = document.getElementById('used');

        const minDisplay = document.getElementById('min');

        const maxDisplay = document.getElementById('max');

        const rangeDisplay = document.getElementById('range');

        const messageDisplay = document.getElementById('message');

        const hintDisplay = document.getElementById('hint');

        const guessesContainer = document.getElementById('guessesContainer');

        const levelButtons = document.querySelectorAll('.level-btn');

 

        // 初始化游戏

        function initGame() {

            // 生成随机目标数字

            targetNumber = Math.floor(Math.random() * 100) + 1;

            console.log(`目标数字: ${targetNumber}`); // 调试用，实际游戏中应移除

           

            // 根据难度设置尝试次数

            attemptsLeft = difficultySettings[difficulty];

            attemptsUsed = 0;

            gameOver = false;

            guessHistory = [];

            minRange = 1;

            maxRange = 100;

           

            // 更新显示

            updateDisplays();

            clearGuesses();

           

            // 设置输入框和按钮状态

            guessInput.value = '';

            guessInput.disabled = false;

            submitBtn.disabled = false;

            guessInput.focus();

           

            // 设置消息

            messageDisplay.textContent = `新游戏开始！猜一个${minRange}到${maxRange}之间的数字`;

            messageDisplay.style.color = '#2d3748';

            hintDisplay.textContent = `你有 ${attemptsLeft} 次机会猜中数字`;

           

            // 更新范围显示

            minDisplay.textContent = minRange;

            maxDisplay.textContent = maxRange;

            rangeDisplay.textContent = `${minRange} - ${maxRange}`;

        }

 

        // 更新显示

        function updateDisplays() {

            attemptsDisplay.textContent = attemptsLeft;

            usedDisplay.textContent = attemptsUsed;

            minDisplay.textContent = minRange;

            maxDisplay.textContent = maxRange;

            rangeDisplay.textContent = `${minRange} - ${maxRange}`;

        }

 

        // 清除猜测历史显示

        function clearGuesses() {

            guessesContainer.innerHTML = '';

        }

 

        // 处理猜测

        function handleGuess() {

            if (gameOver) return;

           

            const guess = parseInt(guessInput.value);

           

            // 验证输入

            if (isNaN(guess) || guess < 1 || guess > 100) {

                messageDisplay.textContent = '请输入1到100之间的有效数字！';

                messageDisplay.style.color = '#e53e3e';

                guessInput.value = '';

                guessInput.focus();

                return;

            }

           

            // 检查是否已经猜过这个数字

            if (guessHistory.includes(guess)) {

                messageDisplay.textContent = `你已经猜过 ${guess} 了，试试别的数字！`;

                messageDisplay.style.color = '#dd6b20';

                guessInput.value = '';

                guessInput.focus();

                return;

            }

           

            // 增加已用尝试次数，减少剩余尝试次数

            attemptsUsed++;

            attemptsLeft--;

           

            // 添加猜测到历史

            guessHistory.push(guess);

           

            // 更新显示

            updateDisplays();

           

            // 显示猜测

            displayGuess(guess);

           

            // 检查猜测是否正确

            if (guess === targetNumber) {

                // 猜对了

                gameWin();

                return;

            }

           

            // 更新范围

            if (guess < targetNumber && guess > minRange) {

                minRange = guess + 1;

                hintDisplay.textContent = `提示：数字比 ${guess} 大`;

            } else if (guess > targetNumber && guess < maxRange) {

                maxRange = guess - 1;

                hintDisplay.textContent = `提示：数字比 ${guess} 小`;

            }

           

            // 检查是否还有尝试次数

            if (attemptsLeft <= 0) {

                gameLose();

                return;

            }

           

            // 更新消息

            if (guess < targetNumber) {

                messageDisplay.textContent = `太小了！数字比 ${guess} 大`;

                messageDisplay.style.color = '#2b6cb0';

            } else {

                messageDisplay.textContent = `太大了！数字比 ${guess} 小`;

                messageDisplay.style.color = '#d69e2e';

            }

           

            // 清空输入框并聚焦

            guessInput.value = '';

            guessInput.focus();

        }

 

        // 显示猜测

        function displayGuess(guess) {

            const guessElement = document.createElement('div');

            guessElement.classList.add('guess');

            guessElement.textContent = guess;

           

            // 根据猜测结果添加样式

            if (guess === targetNumber) {

                guessElement.classList.add('correct');

            } else if (guess < targetNumber) {

                guessElement.classList.add('low');

            } else {

                guessElement.classList.add('high');

            }

           

            guessesContainer.appendChild(guessElement);

           

            // 滚动到最新猜测

            guessElement.scrollIntoView({ behavior: 'smooth', block: 'nearest' });

        }

 

        // 游戏胜利

        function gameWin() {

            gameOver = true;

            messageDisplay.textContent = `恭喜！你猜对了！数字就是 ${targetNumber}`;

            messageDisplay.style.color = '#38a169';

            hintDisplay.textContent = `你用了 ${attemptsUsed} 次尝试，还剩下 ${attemptsLeft} 次机会`;

           

            // 禁用输入和提交按钮

            guessInput.disabled = true;

            submitBtn.disabled = true;

           

            // 显示胜利动画

            const correctGuess = document.querySelector('.guess.correct');

            if (correctGuess) {

                correctGuess.style.animation = 'pulse 1.5s infinite';

            }

        }

 

        // 游戏失败

        function gameLose() {

            gameOver = true;

            messageDisplay.textContent = `游戏结束！正确答案是 ${targetNumber}`;

            messageDisplay.style.color = '#e53e3e';

            hintDisplay.textContent = '点击"重新开始"按钮再试一次！';

           

            // 禁用输入和提交按钮

            guessInput.disabled = true;

            submitBtn.disabled = true;

        }

 

        // 提供提示

        function provideHint() {

            if (gameOver) {

                hintDisplay.textContent = '游戏已结束，请重新开始获取提示';

                return;

            }

           

            // 根据当前范围提供提示

            const range = maxRange - minRange;

            let hint;

           

            if (range <= 10) {

                hint = `数字非常接近！范围现在只有 ${minRange} 到 ${maxRange}`;

            } else if (range <= 25) {

                hint = `范围已经缩小到 ${minRange} 到 ${maxRange}`;

            } else {

                // 提示数字是奇数还是偶数

                hint = targetNumber % 2 === 0

                    ? `提示：目标数字是偶数`

                    : `提示：目标数字是奇数`;

            }

           

            hintDisplay.textContent = hint;

           

            // 添加一个视觉反馈

            hintBtn.style.transform = 'scale(0.95)';

            setTimeout(() => {

                hintBtn.style.transform = 'scale(1)';

            }, 200);

        }

 

        // 切换难度

        function setDifficulty(level) {

            difficulty = level;

           

            // 更新活动按钮

            levelButtons.forEach(btn => {

                if (btn.dataset.level === level) {

                    btn.classList.add('active');

                } else {

                    btn.classList.remove('active');

                }

            });

           

            // 重新开始游戏

            initGame();

        }

 

        // 事件监听器

        submitBtn.addEventListener('click', handleGuess);

        restartBtn.addEventListener('click', initGame);

        hintBtn.addEventListener('click', provideHint);

       

        // 输入框回车键支持

        guessInput.addEventListener('keypress', function(e) {

            if (e.key === 'Enter') {

                handleGuess();

            }

        });

       

        // 难度选择按钮

        levelButtons.forEach(btn => {

            btn.addEventListener('click', function() {

                setDifficulty(this.dataset.level);

            });

        });

 

        // 初始化游戏

        initGame();

    </script>
